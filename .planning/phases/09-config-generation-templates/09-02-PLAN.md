**Change 2: Rewrite _render_nginx_conf() for conditional injection**

Replace the existing _render_nginx_conf() function with this implementation:

```bash
_render_nginx_conf() {
    local template="${SCRIPT_DIR}/templates/nginx.conf.template"
    local output="${AGMIND_DIR}/nginx.conf"

    if [[ ! -f "$template" ]]; then
        die "Template not found: ${template}" "Check templates/ directory for nginx.conf.template"
    fi

    sudo cp "$template" "$output"
    sudo chown "$(whoami)" "$output"

    # Conditionally inject Open Notebook upstream + location
    if [ "${WIZARD_OPEN_NOTEBOOK}" = "1" ]; then
        local nb_upstream
        nb_upstream='upstream open-notebook-app {\n    server open-notebook:8502;\n}'
        _atomic_sed "s|# {{OPTIONAL_UPSTREAMS}}|${nb_upstream}\n# {{OPTIONAL_UPSTREAMS}}|" "$output"

        local nb_location
        nb_location='    # Open Notebook (Streamlit -- WebSocket required)\n    location /notebook/ {\n        proxy_pass http://open-notebook-app/;\n        proxy_http_version 1.1;\n        proxy_set_header Upgrade $http_upgrade;\n        proxy_set_header Connection "upgrade";\n        proxy_set_header Host $host;\n        proxy_set_header X-Real-IP $remote_addr;\n        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;\n        proxy_set_header X-Forwarded-Proto $scheme;\n    }'
        _atomic_sed "s|# {{OPTIONAL_LOCATIONS}}|${nb_location}\n# {{OPTIONAL_LOCATIONS}}|" "$output"
    fi

    # Conditionally inject DB-GPT upstream + location
    if [ "${WIZARD_DBGPT}" = "1" ]; then
        local gpt_upstream
        gpt_upstream='upstream dbgpt-app {\n    server dbgpt:5670;\n}'
        _atomic_sed "s|# {{OPTIONAL_UPSTREAMS}}|${gpt_upstream}\n# {{OPTIONAL_UPSTREAMS}}|" "$output"

        local gpt_location
        gpt_location='    # DB-GPT\n    location /dbgpt/ {\n        proxy_pass http://dbgpt-app/;\n        proxy_http_version 1.1;\n        proxy_set_header Upgrade $http_upgrade;\n        proxy_set_header Connection "upgrade";\n        proxy_set_header Host $host;\n        proxy_set_header X-Real-IP $remote_addr;\n        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;\n        proxy_set_header X-Forwarded-Proto $scheme;\n    }'
        _atomic_sed "s|# {{OPTIONAL_LOCATIONS}}|${gpt_location}\n# {{OPTIONAL_LOCATIONS}}|" "$output"
    fi

    # Remove remaining markers (clean up)
    _atomic_sed "s|# {{OPTIONAL_UPSTREAMS}}||" "$output"
    _atomic_sed "s|# {{OPTIONAL_LOCATIONS}}||" "$output"

    log_info "Generated ${output}"
}
```

IMPORTANT: The sed replacement strings use `\n` for newlines. Verify that _atomic_sed (BSD sed -i '') handles `\n` correctly. If BSD sed does not expand `\n` in replacement, use a different approach: write the nginx blocks to temp files and use sed `r` command to insert, or use printf with actual newlines. The executor MUST test this on macOS.

ALTERNATIVE APPROACH if sed \n does not work on BSD: Instead of sed substitution with \n, use this pattern:
1. Write the upstream/location blocks to a temp file
2. Use sed to find the marker line number
3. Use sed to read the temp file after that line: `sed -i '' "${line_num}r ${tmpfile}" "$output"`
4. Then remove the marker line

The executor should choose whichever approach works reliably with BSD sed on macOS. The key contract is: markers get replaced with correct nginx config blocks.
