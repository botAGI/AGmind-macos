```bash
compose)
    if [ "${2:-}" = "version" ]; then
      # existing version logic stays
      ...
    fi
    # NEW: compose ps with various flags
    if [ "${2:-}" = "ps" ]; then
      case "${3:-}" in
        --status)
          # MOCK_COMPOSE_RUNNING_COUNT controls how many "running" containers
          local _running="${MOCK_COMPOSE_RUNNING_COUNT:-0}"
          if [ "${4:-}" = "running" ] && [ "${5:-}" = "-q" ]; then
            local i=0
            while [ "$i" -lt "$_running" ]; do
              echo "container_${i}"
              i=$((i + 1))
            done
          elif [ "${4:-}" = "exited" ] && [ "${5:-}" = "-q" ]; then
            local _exited="${MOCK_COMPOSE_EXITED_COUNT:-0}"
            local i=0
            while [ "$i" -lt "$_exited" ]; do
              echo "exited_${i}"
              i=$((i + 1))
            done
          fi
          exit 0
          ;;
        -q)
          # Return container IDs for a specific service
          # MOCK_COMPOSE_CONTAINER_ID overrides
          echo "${MOCK_COMPOSE_CONTAINER_ID:-mock_container_abc123}"
          exit 0
          ;;
        --services)
          # MOCK_COMPOSE_SERVICES="api db_postgres redis nginx open-webui"
          local _services="${MOCK_COMPOSE_SERVICES:-api db_postgres redis}"
          for svc in $_services; do echo "$svc"; done
          exit 0
          ;;
        *)
          echo "mock: docker compose ps $*"
          exit 0
          ;;
      esac
    fi
    # NEW: compose up
    if [ "${2:-}" = "up" ]; then
      echo "mock: docker compose up ${3:-}"
      exit 0
    fi
    echo "mock: docker compose $*"
    exit 0
    ;;
```
