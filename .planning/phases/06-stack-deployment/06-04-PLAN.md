**Create tests/helpers/bin/launchctl:**
```bash
#!/bin/bash
# Mock launchctl for testing LaunchAgent management
# MOCK_LAUNCHCTL_LOADED="com.agmind.backup,com.agmind.health" (comma-separated loaded agents)
# MOCK_LAUNCHCTL_BOOTSTRAP="ok"|"fail" (default: ok)
# MOCK_LAUNCHCTL_LOAD="ok"|"fail" (default: ok)
_loaded="${MOCK_LAUNCHCTL_LOADED:-}"
_bootstrap="${MOCK_LAUNCHCTL_BOOTSTRAP:-ok}"
_load="${MOCK_LAUNCHCTL_LOAD:-ok}"

case "$1" in
  list)
    # Check if specific label is loaded
    if [ -n "${2:-}" ]; then
      case ",${_loaded}," in
        *",${2},"*)
          echo "{ \"Label\" = \"${2}\"; }"
          exit 0
          ;;
        *)
          exit 1
          ;;
      esac
    else
      echo "mock: launchctl list"
      exit 0
    fi
    ;;
  bootstrap)
    # launchctl bootstrap gui/<uid> <plist>
    if [ "$_bootstrap" = "ok" ]; then
      echo "mock: launchctl bootstrap $*"
      exit 0
    else
      exit 1
    fi
    ;;
  load)
    # launchctl load <plist>
    if [ "$_load" = "ok" ]; then
      echo "mock: launchctl load $*"
      exit 0
    else
      exit 1
    fi
    ;;
  bootout|unload)
    echo "mock: launchctl $*"
    exit 0
    ;;
  *)
    echo "mock: launchctl $*"
    exit 0
    ;;
esac
```

Setup function:
```bash
setup() {
    # Parent setup (AGMIND_DIR in tmpdir, source common.sh)
    export AGMIND_DIR="${BATS_TEST_TMPDIR}/agmind"
    export AGMIND_LOG_DIR="${AGMIND_DIR}/logs"
    export LOG_FILE="${AGMIND_LOG_DIR}/install.log"
    export STATE_FILE="${AGMIND_DIR}/.install-state"
    mkdir -p "$AGMIND_DIR" "$AGMIND_LOG_DIR" "${AGMIND_DIR}/scripts"
    touch "$LOG_FILE" "$STATE_FILE"
    trap - ERR
    source "${PROJECT_ROOT}/lib/common.sh"
    trap - ERR

    # Set up HOME override for LaunchAgents (don't touch real ~/Library)
    export HOME="${BATS_TEST_TMPDIR}/home"
    mkdir -p "${HOME}/Library/LaunchAgents"

    # SCRIPT_DIR for template paths
    export SCRIPT_DIR="${PROJECT_ROOT}"

    # Source module under test
    source "${PROJECT_ROOT}/lib/backup.sh"
    trap - ERR

    # Default mock config
    export MOCK_PLUTIL_LINT="ok"
    export MOCK_LAUNCHCTL_LOADED=""
    export MOCK_LAUNCHCTL_BOOTSTRAP="ok"

    # Override sudo to be a no-op (scripts use sudo cp)
    sudo() { command "$@"; }
    export -f sudo
}
```
